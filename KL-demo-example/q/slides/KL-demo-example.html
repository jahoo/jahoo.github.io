viewof PriorButtons = Inputs.button([
  ["constant", () => set_(viewof input_prior, Array(dim).fill(1/dim))],
  ["rand. unif.", () => {return set_(viewof input_prior, generate_unif(dim))}],
  ["rand. logunif.", () => {
    return set_(viewof input_prior, generate_logunif(dim, options.probMin == 0 ? 1e-5 : options.probMin, 1))
  }],
  ["rand. determ.", () => {
    let j = Math.floor(Math.random() * dim);
    return set_(viewof input_prior, Array.from({length: dim}, (_,i) => (i===j?1:0)))
  }]
])
